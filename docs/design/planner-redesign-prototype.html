document.querySelectorAll('.day-col').forEach(col => {
  col.addEventListener('click', () => {
    document.querySelectorAll('.day-col')
      .forEach(c => c.classList.remove('selected'));
    col.classList.add('selected');
    renderEventList(col.dataset.date);
  });
});
